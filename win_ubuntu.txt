1 samba配置
sudo apt install samnba
cd /etc/samba
chmod 777 smb.conf
vim smb.conf

打开的配置文件的末尾加上如下内容：
[Ubuntu_18.04]          //映射网络驱动时候用的域名
    comment = Samba
    path = /home/username
    public = yes
    writable = yes
    available = yes
    browseable = yes
    valid users = username

sudo smbpasswd -a username
sudo systemctl restart smbd.service     //重启samba

3 window 映射网络驱动
    文件夹名： \\Ubuntu的IP\映射网络驱动时候用的域名
    eg：\\192.168.125.130\Ubuntu_18.04
